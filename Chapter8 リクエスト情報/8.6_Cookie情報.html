8.6_Cookie情報

Cookieの設定


<?php
setcookie('key1', 'val1');
setcookie('key2', 'val2', time() + 60 * 60);
setcookie('key3', 'val3', time() + 60 * 60 * 24, 'localhost');
setcookie('key4', 'val4', time() + 60 * 60, 'localhost', '/selfphp/');
setcookie('key5', 'val5', time() + 60 * 60, 'localhost', '/selfphp/', secure: true, httponly: true);
