4.1_条件分岐

多岐分岐


<?php
$x = 11;
if ($x >= 20) {
    print 'xは20以上';
} elseif ($x >= 10) {
    print 'xは10以上';
} else {
    print 'xは10未満';
}
?>

実行結果


xは10以上

Switch文


<?php
$x = 2;
switch ($x) {
    case 1:
        print 'rank1';
        break;
    case 2:
        print 'rank2';
    case 3:
        print 'rank3';
        break;
    default:
        print 'default';
}
?>

実行結果


rank2rank3

Switch文の判定方法


<?php
$exp = '3E2';
switch ($exp) {
  case 300 :
    print '値は300';
    break;
  case '3E2' :
    print '値は3E2';
    break;
}
?>

実行結果


値は300

Match式


<?php
$exp = '3E2';
print match($exp) {
  300   => '値は300',
  '3E2' => '値は3E2'
};
?>

実行結果


値は3E2
