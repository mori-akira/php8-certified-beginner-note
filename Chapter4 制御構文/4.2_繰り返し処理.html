4.2_繰り返し処理

while文とdo~while文


<?php
$i = 1;
while ($i < 5) {
    print $i++;
}
do {
    print $i++;
} while ($i < 5);
?>

実行結果


12345

for文


<?php
for ($i = 1, $j = 1; $result = $i * $j, $i < 4; $i++, $j++) {
    print "{$i}×{$j}＝{$result}<br>";
}

実行結果


1×1＝1
2×2＝4
3×3＝9

foreach文

①


<?php
$data = ['高江', '掛谷', '日尾',];
foreach ($data as $value) {
    print "名前は{$value}<br />";
}

実行結果


名前は高江
名前は掛谷
名前は日尾

②


<?php
$data = ['高江' => '男', '掛谷' => '女', '日尾' => '男',];
foreach ($data as $key => $value) {
    print "{$key}：{$value}<br />";
}

実行結果


日尾：男
掛谷：女
日尾：男

値変数の参照渡し


<?php
$data = ['高江', '掛谷', '日尾',];
foreach ($data as &$value) {
  $value = 'New' . $value;
}
print_r($data);

実行結果


Array ( [0] => New高江 [1] => New掛谷 [2] => New日尾 )

foreach文での分割代入


<?php
$data = [['群馬県', '前橋市'], ['栃木県', '宇都宮市'], ['茨城県', '水戸市']];
foreach ($data as [$pref, $city]) {
    print "{$pref}={$city}<br>";
}

実行結果


群馬県=前橋市
栃木県=宇都宮市
茨城県=水戸市

配列への代入へ応用


<?php
$datas = [
    ['pref' => '群馬県', 'city' => '前橋市'],
    ['pref' => '栃木県', 'city' => '宇都宮市'],
    ['pref' => '茨城県', 'city' => '水戸市']
];
$result = [];
foreach ($datas as $data) {
    ['city' => $result[]] = $data;
}
print_r($result);

実行結果


Array ( [0] => 前橋市 [1] => 宇都宮市 [2] => 水戸市 )
