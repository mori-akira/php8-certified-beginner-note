3.2_代入演算子

値渡しと参照渡し


<?php
$x = 1;
$y = $x;
$x = 5;
print $y;
print '<br>';
$x = 1;
$y = &$x;
$x = 5;
print $y;
?>

実行結果


1
5

分割代入

①


<?php
$data = [1, 2, 3, 4, 5];
[$a, $b, $c, $d, $d] = $data;
print $a;
print '<br>';
print $b;
print '<br>';
print $c;
print '<br>';
print $d;
?>

実行結果


1
2
3
5

②


<?php
$data = [1, 2, 3, 4, 5];
[, $b, , $d, $d] = $data;
print $b;
print '<br>';
print $d;
?>

実行結果


2
5

③


<?php
$data = [1, 2, 3, 4, 5];
[1 => $a, 4 => $b] = $data;
print $a;
print '<br>';
print $b;
?>

実行結果


2
5

④


<?php
$data = [1, 2, [3, 4, 5]];
[$_, $_, [$a, $b, $c]] = $data;
print $a;
print '<br>';
print $b;
print '<br>';
print $c;
?>

実行結果


3
4
5

⑤


<?php
[$x, $y] = [3, 5];
[$x, $y] = [$y, $x];
print $x;
print $y;
?>

実行結果


53
